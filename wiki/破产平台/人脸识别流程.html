                # 人脸识别流程

                目前破产平台人脸识别使用微信小程序人脸识别

                # 人脸识别入口

                1. 发起转账提交（对外转账，批量转账）
                2. 转账复核
                3. 债权人会议实名入会
                4. 账号实名认证（saas 平台暂未开启）
                5. 债权人签署债权人查询管理人银行账户流水承诺书

                # 发起人脸识别

                # 1. 获取微信小程序信息

                根据 fydm 获取小程序信息

                export function getWxInfoByFydm(params) {
  return $get("/rlsb/getWxInfoByFydm", params);
}

                # 2. 获取 WxToken

                根据获取到小程序的 appId 获取 accessToken

                export function getWxToken(params) {
  // 获取access_token
  return $post("/rlsb/getWxToken", params);
}

                # 3. 创建二维码

                使用 accessToken 调用微信接口获取调整小程序的二维码

                export function getQrcode(params) {
  // 获取二维码
  return $postArraybuffer("/rlsb/createwxaqrcode", params);
}

                # 进行人脸识别

                # 1. 微信扫码

                使用微信扫描生成的二维码

                扫码后调整到小程序人脸识别页面

                核对个人信息后点击实名认证，实名认证完成后调用接口将认证结果存到缓存中

                wx.request({
  url: "/common/authenticationAddRedis",
  method: "POST",
});

                # 获取认证结果

                web 端创建二维码后将一直轮询查询认证结果

                export function getRedis(params) {
  // 获取实名认证结果
  return $get("/common/getRedis", params);
}
